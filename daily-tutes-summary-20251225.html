import os from serpapi import GoogleSearch def search_web(query: str, num_results: int = 5) -> str: """ Search Google and return formatted results. Args: query: Search query string num_results: Number of results to return Returns: Formatted string with search results """ params = { "q": query, "api_key": os.getenv("SERPAPI_API_KEY"), "num": num_results } search = GoogleSearch(params) results = search.get_dict() # Format results for LLM consumption formatted = [] for r in results.get("organic_results", []): formatted.append(f""" Title: {r.get('title', 'N/A')} URL: {r.get('link', 'N/A')} Summary: {r.get('snippet', 'N/A')} """) return "\n---\n".join(formatted) # Use as agent tool agent = Agent( model=OpenAIChat(id="gpt-4o"), tools=[search_web] )

Why DuckDB for AI Data Agents?

Query Files Directly â†’ No import step needed

Blazing Fast â†’ Columnar storage for fast aggregations

LLM-Friendly â†’ LLMs generate SQL well, DuckDB executes it
